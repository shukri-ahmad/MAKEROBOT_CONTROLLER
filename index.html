let device, characteristic; let transmitInterval; let isBusy = false; // New flag to prevent GATT errors const SEND_RATE = 150; // Increased slightly for stability async function connectBT() { const serviceUuid = "6e400001-b5a3-f393-e0a9-e50e24dcca9e"; const txUuid = "6e400003-b5a3-f393-e0a9-e50e24dcca9e"; try { device = await navigator.bluetooth.requestDevice({ filters: [{ namePrefix: 'BBC micro:bit' }], optionalServices: [serviceUuid] }); const server = await device.gatt.connect(); const service = await server.getPrimaryService(serviceUuid); characteristic = await service.getCharacteristic(txUuid); document.getElementById('status').innerText = "Status: Connected!"; } catch (error) { console.log(error); } } async function sendData(data) { // If Bluetooth is still busy with the last message, skip this one if (!characteristic || isBusy) return; try { isBusy = true; // Block other sends const encoder = new TextEncoder(); await characteristic.writeValue(encoder.encode(data)); } catch (error) { console.log("Send Error:", error); } finally { isBusy = false; // Always unblock, even if it fails } } function startAction(cmd) { if (transmitInterval) clearInterval(transmitInterval); sendData(cmd); transmitInterval = setInterval(() => { sendData(cmd); }, SEND_RATE); } function stopAction() { clearInterval(transmitInterval); transmitInterval = null; // We use a small timeout for the Stop command to ensure // the very last "Move" command has cleared the GATT buffer. setTimeout(() => { sendData('S'); }, 50); } function setupButton(id, command) { const btn = document.getElementById(id); if(!btn) return; btn.addEventListener('mousedown', () => startAction(command)); btn.addEventListener('mouseup', stopAction); btn.addEventListener('mouseleave', stopAction); btn.addEventListener('touchstart', (e) => { e.preventDefault(); startAction(command); }, { passive: false }); btn.addEventListener('touchend', stopAction); } // Initialize setupButton('btn-F', 'F'); setupButton('btn-B', 'B'); setupButton('btn-L', 'L'); setupButton('btn-R', 'R'); setupButton('btn-S', 'S');
