let device, characteristic; let transmitInterval; let isBusy = false; // The "Gatekeeper" flag const SEND_RATE = 150; // Slightly slower for better stability // --- Bluetooth Connection --- async function connectBT() { const serviceUuid = "6e400001-b5a3-f393-e0a9-e50e24dcca9e"; const txUuid = "6e400003-b5a3-f393-e0a9-e50e24dcca9e"; try { device = await navigator.bluetooth.requestDevice({ filters: [{ namePrefix: 'BBC micro:bit' }], optionalServices: [serviceUuid] }); const server = await device.gatt.connect(); const service = await server.getPrimaryService(serviceUuid); characteristic = await service.getCharacteristic(txUuid); document.getElementById('status').innerText = "Status: Connected!"; } catch (error) { console.error("Connection failed:", error); } } // --- High-Performance Send Function --- async function sendData(data) { // If we are already sending or not connected, exit immediately if (!characteristic || isBusy) return; try { isBusy = true; // Lock the gate const encoder = new TextEncoder(); await characteristic.writeValue(encoder.encode(data)); } catch (error) { console.warn("Skipped frame due to GATT congestion."); } finally { isBusy = false; // Unlock the gate for the next command } } // --- Control Logic --- function startAction(cmd) { if (transmitInterval) clearInterval(transmitInterval); sendData(cmd); // Send first burst immediately transmitInterval = setInterval(() => { sendData(cmd); }, SEND_RATE); } function stopAction() { clearInterval(transmitInterval); transmitInterval = null; // Give the GATT queue a tiny 50ms rest before sending the Stop command setTimeout(() => { sendData('S'); }, 50); } // --- Event Listeners --- function setupButton(id, command) { const btn = document.getElementById(id); if (!btn) return; // Mouse Events btn.addEventListener('mousedown', () => startAction(command)); btn.addEventListener('mouseup', stopAction); btn.addEventListener('mouseleave', stopAction); // Touch Events btn.addEventListener('touchstart', (e) => { e.preventDefault(); startAction(command); }, { passive: false }); btn.addEventListener('touchend', stopAction); } // Map your buttons setupButton('btn-F', 'F'); setupButton('btn-B', 'B'); setupButton('btn-L', 'L'); setupButton('btn-R', 'R'); setupButton('btn-S', 'S');
